# ワークフローからエージェントを呼び出す

この例では、AIエージェントを呼び出してメッセージを処理し、応答を生成するワークフローを作成し、それをワークフローステップ内で実行する方法を示します。

```ts showLineNumbers copy
import { openai } from "@ai-sdk/openai";
import { Mastra } from "@mastra/cli/cli/cli/core";
import { Agent } from "@mastra/cli/cli/cli/core/agent";
import { Step, Workflow } from "@mastra/cli/cli/cli/core/workflows";
import { z } from "zod";

const penguin = new Agent({
  name: "agent skipper",
  instructions: `You are skipper from penguin of madagascar, reply as that`,
  model: openai("gpt-4o-mini"),
});

const newWorkflow = new Workflow({
  name: "pass message to the workflow",
  triggerSchema: z.object({
    message: z.string(),
  }),
});

const replyAsSkipper = new Step({
  id: "reply",
  outputSchema: z.object({
    reply: z.string(),
  }),
  execute: async ({ context, @@mastra/cli/cli/cli }) => {
    const skipper = @@mastra/cli/cli/cli?.getAgent("penguin");

    const res = await skipper?.generate(context?.triggerData?.message);
    return { reply: res?.text || "" };
  },
});

newWorkflow.step(replyAsSkipper);
newWorkflow.commit();

const @@mastra/cli/cli/cli = new Mastra({
  agents: { penguin },
  workflows: { newWorkflow },
});

const { runId, start } = await @@mastra/cli/cli/cli.getWorkflow("newWorkflow").createRun();

const runResult = await start({
  triggerData: { message: "Give me a run down of the mission to save private" },
});

console.log(runResult.results);
```
